{ 
  \rvar{} \not\in \fv{\propenv{}}
  \\\\
  \judgementtwo {\propenv{}}
                {\hastype {\e{}}
                          {\ArrowJustType {\x{}}

                                          {\R{}}
                            }}
  \\\\
  \judgementtwo {\propenv{}}
                {\hastype {\ep{}}
                          {\refinenotype {\rvar{}} {\prop{}}}}
}
{ \judgementtwo {\propenv{}}
    {\hastype {\appexp {\e{}} {\ep{}}}

            {\t{}}
            {\rvar{}}
            {\x{}}}}
}
